def wear(smell)
  sample :drum_tom_lo_hard, rate:(smell)
end

def wash(temp)
  play temp, amp: 0.1
end

def dry(dryer)
  sample dryer
  sleep 0.5
end

def dirty?
  1 == [1,2].choose
end

def treat_stains
  sample :bd_808
  sleep 0.5
end

5.times do
  wear(1)
  treat_stains if dirty?
  wash(60)
  dry(:elec_beep)
end

5.times do
  wear(5)
  treat_stains if dirty?
  wash(90)
  dry(:elec_cymbal)
end
